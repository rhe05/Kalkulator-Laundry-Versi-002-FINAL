  function refreshDashboard() {
    if (!document.getElementById("dashboardCabangSummary")) return;

    setDashboardAllLoading();
    google.script.run
      .withSuccessHandler(function (res) {
        if (res && res.ok) {
          cabangListCache = res.data || [];
          ensureDashboardOutletSelection_();
        }
      })
      .withFailureHandler(function () {})
      .listCabang();


    var cId = dashboardSelectedOutletId || null;
    loadDashboardCabang(cId);
    loadDashboardMasterBiaya(cId);
    loadDashboardHPP(cId);
    loadDashboardHargaLayanan(cId);
    loadDashboardFixedCost(cId);
  }
